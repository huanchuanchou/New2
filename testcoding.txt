string food = "ramen";

Random random = new Random();
int i = random.Next(0, 3);

switch (i)
{
  case 0;
      food = "ramen";
      break;
  case 1;
      food = "spaghetti";
      break;
  case 2;
      food = "salad";
      break;
}
console.writeLine("Random recommendation; " + food + ".");

console.writeLine("We have ramen, spaghetti and salad. Which do you like?");
food = Console.ReadLine();
